import matplotlib.pyplot as plt

def visualize_attention_weights(model, images, labels, save_path):
    """
    å¯è§†åŒ–ä¸åŒå›¾åƒçš„æ³¨æ„åŠ›æƒé‡
    """
    model.eval()
    attention_weights_list = []

            # ä»HyperNetä¸­æå–æ³¨æ„åŠ›æƒé‡

    # ç»˜åˆ¶çƒ­åŠ›å›¾

        axes[0, i].imshow(img.permute(1,2,0).cpu().numpy())
        axes[0, i].set_title(f'Score: {label:.2f}')
        axes[0, i].axis('off')
        
        # ä¸‹è¡Œ: æ³¨æ„åŠ›æƒé‡æŸ±çŠ¶å›¾
        bars = axes[1, i].bar(['Stage 0', 'Stage 1', 'Stage 2', 'Stage 3'], weights)

    plt.close()
